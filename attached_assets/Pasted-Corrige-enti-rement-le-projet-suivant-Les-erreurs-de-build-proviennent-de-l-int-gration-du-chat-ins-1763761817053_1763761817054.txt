import { useState, useEffect } from "react";
import { socket } from "@/utils/socket";

export default function ChatBox() {
  const [messages, setMessages] = useState([]);
  const [input, setInput] = useState("");

  useEffect(() => {
    socket.on("receive_message", (msg) => {
      setMessages((prev) => [...prev, msg]);
    });

    return () => {
      socket.off("receive_message");
    };
  }, []);

  const sendMessage = () => {
    if (!input.trim()) return;
    socket.emit("send_message", { text: input, date: new Date() });
    setInput("");
  };

  return (
    <div className="p-4 bg-white rounded-xl shadow border w-full">
      <div className="h-64 overflow-y-auto border-b mb-3">
        {messages.map((m, i) => (
          <div key={i} className="py-1">{m.text}</div>
        ))}
      </div>

      <div className="flex gap-2">
        <input
          value={input}
          onChange={(e) => setInput(e.target.value)}
          className="flex-1 border rounded px-3 py-2"
        />
        <button onClick={sendMessage} className="bg-indigo-600 text-white px-4 rounded">
          Envoyer
        </button>
      </div>
    </div>
  );
}
